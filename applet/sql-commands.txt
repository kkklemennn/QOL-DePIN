
CREATE TABLE devices_binding (
    id BIGINT,
    device_id TEXT,
    owner_address TEXT
);


CREATE TABLE data_table (
    id BIGINT,
    device_id TEXT,
    timestamp TEXT,
    temperature REAL,
    humidity REAL,
    lat DOUBLE PRECISION,
    lon DOUBLE PRECISION,
    accuracy DOUBLE PRECISION
);


CREATE TABLE devices_registry (
    id BIGINT,
    device_id TEXT,
    is_registered BOOLEAN,
    is_active BOOLEAN
);
